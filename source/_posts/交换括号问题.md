由于题目要求括号的数量是偶数，因此简化了解题过程。\
原题连接：[交换括号](https://www.nowcoder.com/questionTerminal/fc5c99ce92aa4d429f68783e3e756a77?page=1&onlyReference=false)

```java
import java.util.Scanner;
public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int t = sc.nextInt();
        while (t -- > 0) {
            int n = sc.nextInt();
            String s = sc.next();

            System.out.println(minSwap(s));

        }
    }

    public static long minSwap(String s) {
        char[] chs = s.toCharArray();
        // 最终答案
        long ans = 0;
        // 临时记录变量
        long res = 0;
        // 遍历所有的字符
        for (char c : chs) {
            // 当发现是 ')' 时，res - 1，表示，')' 括号多了一个
            if (c == ')') {
                res--;
            } else { // 当发现时 '(' 时
                // 查看以下，res 是否小于 0，如果是，则表明前面还有未匹配的 ')'，那么就将这个 ')'，交换到最左边。也就是：)))( -> ()))
                if (res < 0)
                    ans -= res;
                // 记录 '('
                res++;
            }
        }
        return ans;
    }
}
```
